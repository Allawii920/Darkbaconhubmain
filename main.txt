local Fluent = loadstring(game:HttpGet("https://github.com/dawid-scripts/Fluent/releases/latest/download/main.lua"))()

local Player = game.Players.LocalPlayer

-- Create Window
local Window = Fluent:CreateWindow({
    Title = "darkbacon hub | Key System",
    SubTitle = " [ Made by darkbacon ]",
    TabWidth = 160,
    Size = UDim2.fromOffset(400, 300),
    Acrylic = false,
    Theme = "Darker",
    MinimizeKey = Enum.KeyCode.LeftControl
})

Fluent:Notify({
    Title = "Logged in!",
    Content = "You have Logged in as " .. Player.Name,
    SubContent = "SubContent", -- Optional
    Duration = 5 -- Set to nil to make the notification not disappear
})



getgenv().Key = "IlovemyMom"  -- Correct key
getgenv().KeyInput = ""  -- Initial input value (set to empty string)

function DarkBaconScriptHub()
loadstring(game:HttpsGet('https://raw.githubusercontent.com/Testerr111/Darkbaconhubmain/refs/heads/main/Scripthub/main.txt'))()
    end

function CorrectKeyNotification()
    Fluent:Notify({
        Title = "Correct Key!",
        Content = "You have Entered Correct Key!", 
        SubContent = "SubContent", -- Optional
        Duration = 5 -- Set to nil to make the notification not disappear
    })
end

function IncorrectKeyNotification()
    Fluent:Notify({
        Title = "Incorrect Key!",
        Content = "You have Entered Incorrect Key!", 
        SubContent = "SubContent", -- Optional
        Duration = 5 -- Set to nil to make the notification not disappear
    })
end

-- Create Tabs
local Tabs = { 
    Main = Window:AddTab({ Title = "Main", Icon = "home" })
}

-- Add Input Field
local Input = Tabs.Main:AddInput("Input", {
    Title = "Enter Key",
    Default = "",
    Placeholder = "Placeholder",
    Numeric = false, -- Only allows numbers
    Finished = false, -- Only calls callback when you press enter
    Callback = function(Value)
        -- This is where the input value is updated
        getgenv().KeyInput = Value  -- Directly set the value to KeyInput
    end
})

-- Add Button to Check Key
Tabs.Main:AddButton({
    Title = "Check Key",
    Description = "Check Your Key",
    Callback = function()
        if getgenv().KeyInput == getgenv().Key then
            DarkBaconScriptHub()
            CorrectKeyNotification()
        else
            IncorrectKeyNotification()
        end
    end
})


Tabs.Main:AddButton({
    Title = "Destroy Script",
    Description = "Only Destroy Key system",
    Callback = function()
        Fluent:Destroy()
        end
})
